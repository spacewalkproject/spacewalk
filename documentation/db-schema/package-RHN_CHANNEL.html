    37:     PROCEDURE subscribe_server(server_id_in IN NUMBER, channel_id_in NUMBER, immediate_in NUMBER := 1, user_id_in in number := null, recalcfamily_in number := 1)
    38:     IS
    39:         channel_parent_val      rhnChannel.parent_channel%TYPE;
    40:         parent_subscribed       BOOLEAN;
    41:         server_has_base_chan    BOOLEAN;
    42:         server_already_in_chan  BOOLEAN;
    43:         channel_family_id_val   NUMBER;
    44:         server_org_id_val       NUMBER;
    45:         available_subscriptions NUMBER;
    46:         available_fve_subs      NUMBER;
    47:         consenting_user         NUMBER;
    48:         allowed                 number := 0;
    49:         is_fve                  CHAR(1) := 'N';
    50:     BEGIN
    51:         if user_id_in is not null then
    52:             allowed := rhn_channel.user_role_check(channel_id_in, user_id_in, 'subscribe');
    53:         else
    54:             allowed := 1;
    55:         end if;
    56: 
    57:         if allowed = 0 then
    58:             rhn_exception.raise_exception('no_subscribe_permissions');
    59:         end if;
    60: 
    61: 
    62:         SELECT parent_channel INTO channel_parent_val FROM rhnChannel WHERE id = channel_id_in;
    63: 
    64:         IF channel_parent_val IS NOT NULL
    65:         THEN
    66:             -- child channel; if attempting to cross-subscribe a child to the wrong base, silently ignore
    67:             parent_subscribed := FALSE;
    68: 
    69:             FOR check_subscription IN check_server_subscription(server_id_in, channel_parent_val)
    70:             LOOP
    71:                 parent_subscribed := TRUE;
    72:             END LOOP check_subscription;
    73: 
    74:             IF NOT parent_subscribed
    75:             THEN
    76:                 RETURN;
    77:             END IF;
    78:         ELSE
    79:             -- base channel
    80:             server_has_base_chan := FALSE;
    81:             FOR base IN server_base_subscriptions(server_id_in)
    82:             LOOP
    83:                 server_has_base_chan := TRUE;
    84:             END LOOP base;
    85: 
    86:             IF server_has_base_chan
    87:             THEN
    88:                 rhn_exception.raise_exception('channel_server_one_base');
    89:             END IF;
    90:         END IF;
    91: 
    92:         FOR check_subscription IN check_server_subscription(server_id_in, channel_id_in)
    93:         LOOP
    94:             server_already_in_chan := TRUE;
    95:         END LOOP check_subscription;
    96: 
    97:         IF server_already_in_chan
    98:         THEN
    99:             RETURN;
   100:         END IF;
   101: 
   102:         channel_family_id_val := rhn_channel.family_for_channel(channel_id_in);
   103:         IF channel_family_id_val IS NULL
   104:         THEN
   105:             rhn_exception.raise_exception('channel_subscribe_no_family');
   106:         END IF;
   107: 
   108:         --
   109:         -- Use the org_id of the server only if the org_id of the channel = NULL.
   110:         -- This is required for subscribing to shared channels.
   111:         --
   112:         SELECT NVL(org_id, (SELECT org_id FROM rhnServer WHERE id = server_id_in))
   113:           INTO server_org_id_val
   114:           FROM rhnChannel
   115:          WHERE id = channel_id_in;
   116: 
   117:         begin
   118:             obtain_read_lock(channel_family_id_val, server_org_id_val);
   119:         exception
   120:             when no_data_found then
   121:                 rhn_exception.raise_exception('channel_family_no_subscriptions');
   122:         end;
   123: 
   124:         available_subscriptions := rhn_channel.available_family_subscriptions(channel_family_id_val, server_org_id_val);
   125:         available_fve_subs := rhn_channel.available_fve_family_subs(channel_family_id_val, server_org_id_val);
   126: 
   127:         IF available_subscriptions IS NULL OR
   128:             available_subscriptions > 0 or
   129:             can_server_consume_virt_channl(server_id_in, channel_family_id_val) = 1 OR
   130:             (available_fve_subs > 0 AND can_server_consume_fve(server_id_in) = 1)
   131:         THEN
   132:             if can_server_consume_virt_channl(server_id_in, channel_family_id_val) = 0 AND available_fve_subs > 0 AND can_server_consume_fve(server_id_in) = 1 THEN
   133:                 is_fve := 'Y';
   134:             END IF;
   135:             insert into rhnServerHistory (id,server_id,summary,details) (
   136:                 select  rhn_event_id_seq.nextval,
   137:                         server_id_in,
   138:                         'subscribed to channel ' || SUBSTR(c.label, 0, 106),
   139:                         c.label
   140:                 from    rhnChannel c
   141:                 where   c.id = channel_id_in
   142:             );
   143:             UPDATE rhnServer SET channels_changed = sysdate WHERE id = server_id_in;
   144:             INSERT INTO rhnServerChannel (server_id, channel_id, is_fve) VALUES (server_id_in, channel_id_in, is_fve);
   145: 			IF recalcfamily_in > 0
   146: 			THEN
   147:                 rhn_channel.update_family_counts(channel_family_id_val, server_org_id_val);
   148: 			END IF;
   149:             queue_server(server_id_in, immediate_in);
   150:         ELSE
   151:             rhn_exception.raise_exception('channel_family_no_subscriptions');
   152:         END IF;
   153: 
   154:     END subscribe_server;

   276:     function guess_server_base(
   277:         server_id_in in number
   278:     ) RETURN number is
   279:         cursor server_cursor is
   280:             select s.server_arch_id, s.release, s.org_id
   281:               from rhnServer s
   282:              where s.id = server_id_in;
   283:     begin
   284:         for s in server_cursor loop
   285:             for channel in base_channel_cursor(s.release,
   286:                 s.server_arch_id, s.org_id)
   287:             loop
   288:                 return channel.id;
   289:             end loop base_channel_cursor;
   290:         end loop server_cursor;
   291:         -- Server not found, or no base channel applies to it
   292:         return null;

   295:     -- Private function
   296:     function normalize_server_arch(server_arch_in in varchar2)
   297:     return varchar2
   298:     deterministic
   299:     is
   300:         suffix VARCHAR2(128) := '-redhat-linux';
   301:         suffix_len NUMBER := length(suffix);
   302:     begin
   303:         if server_arch_in is NULL then
   304:             return NULL;
   305:         end if;
   306:         if instr(server_arch_in, '-') > 0
   307:         then
   308:             -- Suffix already present
   309:             return server_arch_in;
   310:         end if;
   311:         return server_arch_in || suffix;

   342:     function base_channel_rel_archid(
   343:         release_in in varchar2,
   344:         server_arch_id_in in number,
   345:         org_id_in in number := -1,
   346:         user_id_in in number := null
   347:     ) return number is
   348:         denied_channel_id number := null;
   349:         valid_org_id number := org_id_in;
   350:         valid_user_id number := user_id_in;
   351:         channel_subscribable number;
   352:     begin
   353:         if org_id_in = -1 and user_id_in is not null then
   354:             -- Get the org id from the user id
   355:             begin
   356:                 select org_id
   357:                   into valid_org_id
   358:                   from web_contact
   359:                  where id = user_id_in;
   360:             exception
   361:                 when no_data_found then
   362:                     -- User doesn't exist
   363:                     -- XXX Only list public stuff for now
   364:                     valid_user_id := null;
   365:                     valid_org_id := -1;
   366:             end;
   367:         end if;
   368: 
   369:         for c in base_channel_cursor(release_in, server_arch_id_in, valid_org_id)
   370:         loop
   371:             -- This row is a possible match
   372:             if valid_user_id is null then
   373:                 -- User ID not specified, so no user to channel permissions to
   374:                 -- check
   375:                 return c.id;
   376:             end if;
   377: 
   378:             -- Check user to channel permissions
   379:             select loose_user_role_check(c.id, user_id_in, 'subscribe')
   380:               into channel_subscribable
   381:               from dual;
   382: 
   383:             if channel_subscribable = 1 then
   384:                 return c.id;
   385:             end if;
   386: 
   387:             -- Base channel exists, but is not subscribable; keep trying
   388:             denied_channel_id := c.id;
   389:         end loop base_channel_fetch;
   390: 
   391:         if denied_channel_id is not null then
   392:             rhn_exception.raise_exception('no_subscribe_permissions');
   393:         end if;
   394:         -- No base channel applies
   395:         return NULL;
   396:     end base_channel_rel_archid;

   400:     IS
   401:         cursor server_channels(server_id_in in number) is
   402:                 select  s.org_id, sc.channel_id, cfm.channel_family_id
   403:                 from    rhnServer s,
   404:                         rhnServerChannel sc,
   405:                         rhnChannelFamilyMembers cfm
   406:                 where   s.id = server_id_in
   407:                         and s.id = sc.server_id
   408:                         and sc.channel_id = cfm.channel_id
   409:                 order by cfm.channel_family_id;
   410:         last_channel_family_id rhnChannelFamilyMembers.channel_family_id%type := -1;
   411:         last_channel_org_id    rhnServer.org_id%type := -1;
   412:     BEGIN
   413:         for channel in server_channels(server_id_in)
   414:         loop
   415:                 unsubscribe_server(server_id_in, channel.channel_id, 1, 1, deleting_server, 0);
   416:                 if update_family_countsYN > 0
   417:                     and channel.channel_family_id != last_channel_family_id then
   418:                     -- update family counts only once
   419:                     -- after all channels with same family has been fetched
   420:                     if last_channel_family_id != -1 then
   421:                         update_family_counts(last_channel_family_id, last_channel_org_id);
   422:                     end if;
   423:                     last_channel_family_id := channel.channel_family_id;
   424:                     last_channel_org_id    := channel.org_id;
   425:                 end if;
   426:         end loop channel;
   427:         if update_family_countsYN > 0 and last_channel_family_id != -1 then
   428:             -- update the last family fetched
   429:             update_family_counts(last_channel_family_id, last_channel_org_id);
   430:         end if;
   431:     END clear_subscriptions;

   433:     PROCEDURE unsubscribe_server(server_id_in IN NUMBER, channel_id_in NUMBER, immediate_in NUMBER := 1, unsubscribe_children_in number := 0,
   434:                                  deleting_server IN NUMBER := 0,
   435:                                  update_family_countsYN IN NUMBER := 1)
   436:     IS
   437:         channel_family_id_val   NUMBER;
   438:         server_org_id_val       NUMBER;
   439:         available_subscriptions NUMBER;
   440:         server_already_in_chan  BOOLEAN;
   441:         cursor  channel_family_is_proxy(channel_family_id_in in number) is
   442:                 select  1
   443:                 from    rhnChannelFamily
   444:                 where   id = channel_family_id_in
   445:                     and label = 'rhn-proxy';
   446:         cursor  channel_family_is_satellite(channel_family_id_in in number) is
   447:                 select  1
   448:                 from    rhnChannelFamily
   449:                 where   id = channel_family_id_in
   450:                     and label = 'rhn-satellite';
   451:         -- this is *EXACTLY* like check_server_parent_membership, but if we recurse
   452:         -- with the package-level one, we get a "cursor already open", so we need a
   453:         -- copy on our call stack instead.  GROAN.
   454:         cursor local_chk_server_parent_memb (
   455:                         server_id_in number,
   456:                         channel_id_in number ) is
   457:                 select  c.id
   458:                 from    rhnChannel                      c,
   459:                                 rhnServerChannel        sc
   460:                 where   1=1
   461:                         and c.parent_channel = channel_id_in
   462:                         and c.id = sc.channel_id
   463:                         and sc.server_id = server_id_in;
   464:     BEGIN
   465:         FOR child IN local_chk_server_parent_memb(server_id_in, channel_id_in)
   466:         LOOP
   467:             if unsubscribe_children_in = 1 then
   468:                 unsubscribe_server(server_id_in => server_id_in,
   469:                                                                 channel_id_in => child.id,
   470:                                                                 immediate_in => immediate_in,
   471:                                                                 unsubscribe_children_in => unsubscribe_children_in,
   472:                         deleting_server => deleting_server,
   473:                         update_family_countsYN => update_family_countsYN);
   474:             else
   475:                 rhn_exception.raise_exception('channel_unsubscribe_child_exists');
   476:             end if;
   477:         END LOOP child;
   478: 
   479:         server_already_in_chan := FALSE;
   480: 
   481:         FOR check_subscription IN check_server_subscription(server_id_in, channel_id_in)
   482:         LOOP
   483:             server_already_in_chan := TRUE;
   484:         END LOOP check_subscription;
   485: 
   486:         IF NOT server_already_in_chan
   487:         THEN
   488:             RETURN;
   489:         END IF;
   490: 
   491:    if deleting_server = 0 then
   492: 
   493:       insert into rhnServerHistory (id,server_id,summary,details) (
   494:           select  rhn_event_id_seq.nextval,
   495:                 server_id_in,
   496:              'unsubscribed from channel ' || SUBSTR(c.label, 0, 106),
   497:              c.label
   498:           from    rhnChannel c
   499:           where   c.id = channel_id_in
   500:       );
   501: 
   502:         UPDATE rhnServer SET channels_changed = sysdate WHERE id = server_id_in;
   503:    end if;
   504: 
   505:    DELETE FROM rhnServerChannel WHERE server_id = server_id_in AND channel_id = channel_id_in;
   506: 
   507:    if deleting_server = 0 then
   508:         queue_server(server_id_in, immediate_in);
   509:    end if;
   510: 
   511:         channel_family_id_val := rhn_channel.family_for_channel(channel_id_in);
   512:         IF channel_family_id_val IS NULL
   513:         THEN
   514:             rhn_exception.raise_exception('channel_unsubscribe_no_family');
   515:         END IF;
   516: 
   517:         for ignore in channel_family_is_satellite(channel_family_id_val) loop
   518:                 delete from rhnSatelliteInfo where server_id = server_id_in;
   519:         end loop;
   520: 
   521:         for ignore in channel_family_is_proxy(channel_family_id_val) loop
   522:                 delete from rhnProxyInfo where server_id = server_id_in;
   523:         end loop;
   524:         SELECT org_id INTO server_org_id_val
   525:           FROM rhnServer
   526:          WHERE id = server_id_in;
   527: 
   528:         if update_family_countsYN = 1 then
   529:             rhn_channel.update_family_counts(channel_family_id_val, server_org_id_val);
   530:         end if;
   531:     END unsubscribe_server;

   550:     FUNCTION available_family_subscriptions(channel_family_id_in IN NUMBER, org_id_in IN NUMBER)
   551:     RETURN NUMBER
   552:     IS
   553:         cfp channel_family_perm_cursor%ROWTYPE;
   554:         current_members_val NUMBER;
   555:         max_members_val     NUMBER;
   556:         found               NUMBER;
   557:     BEGIN
   558:         IF NOT channel_family_perm_cursor%ISOPEN
   559:         THEN
   560:             OPEN channel_family_perm_cursor(channel_family_id_in, org_id_in);
   561:         END IF;
   562: 
   563:         FETCH channel_family_perm_cursor INTO cfp;
   564: 
   565:         WHILE channel_family_perm_cursor%FOUND
   566:         LOOP
   567:             found := 1;
   568: 
   569:             current_members_val := cfp.current_members;
   570:             max_members_val := cfp.max_members;
   571: 
   572:             FETCH channel_family_perm_cursor INTO cfp;
   573:         END LOOP;
   574: 
   575:         IF channel_family_perm_cursor%ISOPEN
   576:         THEN
   577:             CLOSE channel_family_perm_cursor;
   578:         END IF;
   579: 
   580:         -- not found: either the channel fam doesn't have an entry in cfp, or the org doesn't have access to it.
   581:         -- either way, there are no available subscriptions
   582: 
   583:         IF found IS NULL
   584:         THEN
   585:             RETURN 0;
   586:         END IF;
   587: 
   588:         -- null max members?  in that case, pass it on; NULL means infinite
   589:         IF max_members_val IS NULL
   590:         THEN
   591:             RETURN NULL;
   592:         END IF;
   593: 
   594:         -- otherwise, return the delta
   595:         RETURN max_members_val - current_members_val;
   596:     END available_family_subscriptions;

   598:     FUNCTION available_fve_family_subs(channel_family_id_in IN NUMBER, org_id_in IN NUMBER)
   599:     RETURN NUMBER

   603:         fve_max_members_val     NUMBER;
   604:         found               NUMBER;
   605: 
   606:     BEGIN
   607:         IF NOT channel_family_perm_cursor%ISOPEN THEN
   608:             OPEN channel_family_perm_cursor(channel_family_id_in, org_id_in);
   609:         END IF;
   610: 
   611:         FETCH channel_family_perm_cursor INTO cfp;

   613:         WHILE channel_family_perm_cursor%FOUND LOOP
   614:             found := 1;
   615:             fve_current_members_val := cfp.fve_current_members;

   627:

   629:             RETURN NULL;

   810:     -- check if a user has a given role, or if such a role is inferrable
   811:     function user_role_check_debug(channel_id_in in number,
   812:                                    user_id_in in number,
   813:                                    role_in in varchar2,
   814:                                    reason_out out varchar2)
   815:     return number
   816:     is
   817:         org_id number;
   818:     begin
   819:         org_id := rhn_user.get_org_id(user_id_in);
   820: 
   821:         -- channel might be shared
   822:         if role_in = 'subscribe' and
   823:            rhn_channel.shared_user_role_check(channel_id_in, user_id_in, role_in) = 1 then
   824:             return 1;
   825:         end if;
   826: 
   827:         if role_in = 'manage' and
   828:            NVL(rhn_channel.get_org_id(channel_id_in), -1) <> org_id then
   829:                 reason_out := 'channel_not_owned';
   830:                return 0;
   831:             end if;
   832: 
   833:         if role_in = 'subscribe' and
   834:            rhn_channel.get_org_access(channel_id_in, org_id) = 0 then
   835:                 reason_out := 'channel_not_available';
   836:                 return 0;
   837:             end if;
   838: 
   839:         -- channel admins have all roles
   840:         if rhn_user.check_role_implied(user_id_in, 'channel_admin') = 1 then
   841:             reason_out := 'channel_admin';
   842:             return 1;
   843:             end if;
   844: 
   845:         -- the subscribe permission is inferred
   846:     -- UNLESS the not_globally_subscribable flag is set
   847:         if role_in = 'subscribe'
   848:         then
   849:             if rhn_channel.org_channel_setting(channel_id_in,
   850:                        org_id,
   851:                        'not_globally_subscribable') = 0 then
   852:                 reason_out := 'globally_subscribable';
   853:                     return 1;
   854:             end if;
   855:         end if;
   856: 
   857:         -- all other roles (manage right now) are explicitly granted
   858:         reason_out := 'direct_permission';
   859:         return rhn_channel.direct_user_role_check(channel_id_in,
   860:                                               user_id_in, role_in);
   861:     end;

   948:     FUNCTION channel_priority(channel_id_in IN number)
   949:     RETURN number
   950:     IS
   951:          channel_name varchar2(256);
   952:          priority number;
   953:          end_of_life_val date;
   954:          org_id_val number;
   955:     BEGIN
   956: 
   957:         select name, end_of_life, org_id
   958:         into channel_name, end_of_life_val, org_id_val
   959:         from rhnChannel
   960:         where id = channel_id_in;
   961: 
   962:         if end_of_life_val is not null then
   963:           return -400;
   964:         end if;
   965: 
   966:         if channel_name like 'Red Hat Enterprise Linux%' or channel_name like 'RHEL%' then
   967:           priority := 1000;
   968:           if channel_name not like '%Beta%' then
   969:             priority := priority + 1000;
   970:           end if;
   971: 
   972:           priority := priority +
   973:             case

   168:    function can_entitle_server (
   169:         server_id_in in number,
   170:         type_label_in in varchar2 )
   171:    return number is
   172:       cursor addon_servergroups (base_label_in in varchar2,
   173:                                  addon_label_in in varchar2) is
   174:          select
   175:             addon_id
   176:          from
   177:             rhnSGTypeBaseAddonCompat
   178:          where base_id = lookup_sg_type (base_label_in)
   179:            and addon_id = lookup_sg_type (addon_label_in);
   180: 
   181:       previous_ent        rhn_entitlements.ents_array;
   182:       is_base_in          char   := 'N';
   183:       is_base_current     char   := 'N';
   184:       i                   number := 0;
   185:       sgid                number := 0;
   186: 
   187:    begin
   188: 
   189:       previous_ent := rhn_entitlements.ents_array();
   190:       previous_ent := rhn_entitlements.get_server_entitlement(server_id_in);
   191: 
   192:       select distinct is_base
   193:       into is_base_in
   194:       from rhnServerGroupType
   195:       where label = type_label_in;
   196: 
   197:       if previous_ent.count = 0 then
   198:          sgid := find_compatible_sg (server_id_in, type_label_in);
   199:          if (is_base_in = 'Y' and sgid is not null) then
   200:             -- rhn_server.insert_into_servergroup (server_id_in, sgid);
   201:             return 1;
   202:          else
   203:             -- rhn_exception.raise_exception ('invalid_base_entitlement');
   204:             return 0;
   205:          end if;
   206: 
   207:       -- there are previous ents, first make sure we're not trying to entitle a base ent
   208:       elsif is_base_in = 'Y' then
   209:          -- rhn_exception.raise_exception ('invalid_addon_entitlement');
   210:          return 0;
   211: 
   212:       -- it must be an addon, so proceed with the entitlement
   213:       else
   214: 
   215:          -- find the servers base ent
   216:          is_base_current := 'N';
   217:          i := 0;
   218:          while is_base_current = 'N' and i <= previous_ent.count
   219:          loop
   220:             i := i + 1;
   221:             select is_base
   222:             into is_base_current
   223:             from rhnServerGroupType
   224:             where label = previous_ent(i);
   225:          end loop;
   226: 
   227:          -- never found a base ent, that would be strange
   228:          if is_base_current  = 'N' then
   229:             -- rhn_exception.raise_exception ('invalid_base_entitlement');
   230:             return 0;
   231:          end if;
   232: 
   233:          -- this for loop verifies the validity of the addon path
   234:          for addon_servergroup in addon_servergroups  (previous_ent(i), type_label_in) loop
   235:             -- find an appropriate sgid for the addon and entitle the server
   236:             sgid := find_compatible_sg (server_id_in, type_label_in);
   237:             if sgid is not null then
   238:                -- rhn_server.insert_into_servergroup (server_id_in, sgid);
   239:                return 1;
   240:             else
   241:                -- rhn_exception.raise_exception ('invalid_addon_entitlement');
   242:                return 0;
   243:             end if;
   244:          end loop;
   245: 
   246:       end if;
   247: 
   248:       return 0;
   249: 
   250:    end can_entitle_server;

   484:     -- *******************************************************************
   485:     -- PROCEDURE: repoll_virt_guest_entitlements
   486:     --
   487:     --   Whenever we add/remove a virtualization_host* entitlement from
   488:     --   a host, we can call this procedure to update what type of slots
   489:     --   the guests are consuming.
   490:     --
   491:     --   If you're removing the entitlement, it's
   492:     --   possible the guests will become unentitled if you don't have enough
   493:     --   physical slots to cover them.
   494:     --
   495:     --   If you're adding the entitlement, you end up freeing up physical
   496:     --   slots for other systems.
   497:     --
   498:     -- *******************************************************************
   499:     procedure repoll_virt_guest_entitlements(server_id_in in number)
   500:     is
   501: 
   502:         -- All channel families associated with the guests of server_id_in
   503:         cursor families is
   504:             select distinct cfs.channel_family_id
   505:             from
   506:                 rhnChannelFamilyServers cfs,
   507:                 rhnVirtualInstance vi
   508:             where
   509:                 vi.host_system_id = server_id_in
   510:                 and vi.virtual_system_id = cfs.server_id;
   511: 
   512:         -- All of server group types associated with the guests of
   513:         -- server_id_in
   514:         cursor group_types is
   515:             select distinct sg.group_type, sgt.label
   516:             from
   517:                 rhnServerGroupType sgt,
   518:                 rhnServerGroup sg,
   519:                 rhnServerGroupMembers sgm,
   520:                 rhnVirtualInstance vi
   521:             where
   522:                 vi.host_system_id = server_id_in
   523:                 and vi.virtual_system_id = sgm.server_id
   524:                 and sgm.server_group_id = sg.id
   525:                 and sg.group_type = sgt.id;
   526: 
   527:         -- Virtual servers from a certain family belonging to a speicifc
   528:         -- host that are consuming physical channel slots over the limit.
   529:         cursor virt_servers_cfam(family_id_in in number, quantity_in in number)  is
   530:             select virtual_system_id
   531:             from (
   532:                 select rownum, vi.virtual_system_id
   533:                 from
   534:                     rhnChannelFamilyMembers cfm,
   535:                     rhnServerChannel sc,
   536:                     rhnVirtualInstance vi
   537:                 where
   538:                     vi.host_system_id = server_id_in
   539:                     and vi.virtual_system_id = sc.server_id
   540:                     and sc.channel_id = cfm.channel_id
   541:                     and cfm.channel_family_id = family_id_in
   542:                 order by sc.modified desc
   543:                 )
   544:             where rownum <= quantity_in;
   545: 
   546:         -- Virtual servers from a certain family belonging to a speicifc
   547:         -- host that are consuming physical system slots over the limit.
   548:         cursor virt_servers_sgt(group_type_in in number, quantity_in in number)  is
   549:             select virtual_system_id
   550:             from (
   551:                 select rownum, vi.virtual_system_id
   552:                 from
   553:                     rhnServerGroup sg,
   554:                     rhnServerGroupMembers sgm,
   555:                     rhnVirtualInstance vi
   556:                 where
   557:                     vi.host_system_id = server_id_in
   558:                     and vi.virtual_system_id = sgm.server_id
   559:                     and sgm.server_group_id = sg.id
   560:                     and sg.group_type = group_type_in
   561:                 order by sgm.modified desc
   562:                 )
   563:             where rownum <= quantity_in;
   564:         -- Get the orgs of Virtual guests
   565:         -- Since they may belong to different orgs
   566:         cursor virt_guest_orgs  is
   567:                 select  distinct (s.org_id)
   568:                 from rhnServer s
   569:                     inner join  rhnVirtualInstance vi on vi.virtual_system_id = s.id
   570:                 where
   571:                     vi.host_system_id = server_id_in
   572:                     and s.org_id <> (select s1.org_id from rhnServer s1 where s1.id = vi.host_system_id) ;
   573: 
   574: 
   575:         org_id_val number;
   576:         max_members_val number;
   577:         max_flex_val number;
   578:         current_members_calc number;
   579:         sg_id number;
   580:         is_virt number := 0;
   581:     begin
   582:         begin
   583:           select 1 into is_virt
   584:                 from rhnServerEntitlementView
   585:            where server_id = server_id_in
   586:                  and label in ('virtualization_host', 'virtualization_host_platform');
   587:            exception
   588:                 when no_data_found then
   589:                   is_virt := 0;
   590:         end;
   591: 
   592:         select org_id
   593:         into org_id_val
   594:         from rhnServer
   595:         where id = server_id_in;
   596: 
   597:         -- deal w/ channel entitlements first ...
   598:         for family in families loop
   599:             if is_virt = 0 then
   600:             -- if the host_server does not have virt
   601:             --- find all possible flex slots
   602:             -- and set each of the flex eligible guests to Y
   603:                 UPDATE rhnServerChannel sc set sc.is_fve = 'Y'
   604:                 where sc.server_id in (
   605:                        select virtual_system_id from (
   606:                             select rownum, vi.virtual_system_id,  sfc.max_members - sfc.current_members as free_slots
   607:                             from rhnServerFveCapable sfc
   608:                                 inner join rhnVirtualInstance vi on vi.virtual_system_id = sfc.server_id
   609:                             where vi.host_system_id = server_id_in
   610:                                   and sfc.channel_family_id = family.channel_family_id
   611:                               order by vi.modified desc
   612:                           )
   613:                         where rownum <=  free_slots
   614:                 );
   615:             else
   616:             -- if the host_server has virt
   617:             -- set all its flex guests to N
   618:                 UPDATE rhnServerChannel sc set sc.is_fve = 'N'
   619:                 where
   620:                     sc.channel_id in (select cfm.channel_id from rhnChannelFamilyMembers cfm
   621:                                       where cfm.CHANNEL_FAMILY_ID = family.channel_family_id)
   622:                     and sc.is_fve = 'Y'
   623:                     and sc.server_id in
   624:                             (select vi.virtual_system_id  from rhnVirtualInstance vi
   625:                                     where vi.host_system_id = server_id_in);
   626:             end if;
   627: 
   628:             -- get the current (physical) members of the family
   629:             current_members_calc :=
   630:                 rhn_channel.channel_family_current_members(family.channel_family_id,
   631:                                                            org_id_val); -- fixed transposed args
   632: 
   633:             -- get the max members of the family
   634:             select max_members
   635:             into max_members_val
   636:             from rhnPrivateChannelFamily
   637:             where channel_family_id = family.channel_family_id
   638:             and org_id = org_id_val;
   639: 
   640:             select fve_max_members
   641:             into max_flex_val
   642:             from rhnPrivateChannelFamily
   643:             where channel_family_id = family.channel_family_id
   644:             and org_id = org_id_val;
   645: 
   646:             if current_members_calc > max_members_val then
   647:                 -- A virtualization_host* ent must have been removed, so we'll
   648:                 -- unsubscribe guests from the host first.
   649: 
   650:                 -- hm, i don't think max_members - current_members_calc yielding a negative number
   651:                 -- will work w/ rownum, swaping 'em in the body of this if...
   652:                 for virt_server in virt_servers_cfam(family.channel_family_id,
   653:                                 current_members_calc - max_members_val) loop
   654: 
   655:                     rhn_channel.unsubscribe_server_from_family(
   656:                                 virt_server.virtual_system_id,
   657:                                 family.channel_family_id);
   658:                 end loop;
   659: 
   660:                 -- if we're still over the limit, which would be odd,
   661:                 -- just prune the group to max_members
   662:                 --
   663:                 -- er... wouldn't we actually have to refresh the values of
   664:                 -- current_members_calc and max_members_val to actually ever
   665:                 -- *skip this??
   666:                 if current_members_calc > max_members_val then
   667:                     -- argh, transposed again?!
   668:                     set_family_count(org_id_val,
   669:                                      family.channel_family_id,
   670:                                      max_members_val, max_flex_val);
   671:                     --TODO calculate this correctly
   672:                 end if;
   673: 
   674:            end if;
   675: 
   676:             -- update current_members for the family.  This will set the value
   677:             -- to reflect adding/removing the entitlement.
   678:             --
   679:             -- what's the difference of doing this vs the unavoidable set_family_count above?
   680:             rhn_channel.update_family_counts(family.channel_family_id,
   681:                                              org_id_val);
   682: 
   683:             -- It is possible that the guests belong  to a different org than the host
   684:             -- so we are going to update the family counts in the guests orgs also
   685:             for org in virt_guest_orgs loop
   686:                     rhn_channel.update_family_counts(family.channel_family_id,
   687:                                              org.org_id);
   688:             end loop;
   689:         end loop;
   690: 
   691:         for a_group_type in group_types loop
   692:           -- get the current *physical* members of the system entitlement type for the org...
   693:           --
   694:           -- unlike channel families, it appears the standard rhnServerGroup.max_members represents
   695:           -- *physical* slots, vs physical+virt ... boy that's confusing...
   696: 
   697:           select max_members, id
   698:             into max_members_val, sg_id
   699:             from rhnServerGroup
   700:             where group_type = a_group_type.group_type
   701:             and org_id = org_id_val;
   702: 
   703: 
   704:       select count(sep.server_id) into current_members_calc
   705:             from rhnServerEntitlementPhysical sep
   706:            where sep.server_group_id = sg_id
   707:              and sep.server_group_type_id = a_group_type.group_type;
   708: 
   709:           if current_members_calc > max_members_val then
   710:             -- A virtualization_host* ent must have been removed, and we're over the limit, so unsubscribe guests
   711:             for virt_server in virt_servers_sgt(a_group_type.group_type,
   712:                                                 current_members_calc - max_members_val) loop
   713:               rhn_entitlements.remove_server_entitlement(virt_server.virtual_system_id, a_group_type.label);
   714: 
   715:               -- decrement current_members_calc, we'll use it to reset current_members for the group at the end...
   716:               current_members_calc := current_members_calc - 1;
   717:             end loop;
   718: 
   719:           end if;
   720: 
   721:           update rhnServerGroup set current_members = current_members_calc
   722:            where org_id = org_id_val
   723:              and group_type = a_group_type.group_type;
   724: 
   725:           -- I think that's all the house-keeping we have to do...
   726:         end loop;
   727: 
   728:     end repoll_virt_guest_entitlements;

   968:     -- PROCEDURE: prune_group
   969:     -- Unsubscribes servers consuming physical slots that over the org's
   970:     --   limit.
   971:     -- Called by: set_group_count, repoll_virt_guest_entitlements
   972:     -- *******************************************************************
   973:     procedure prune_group (
   974:         group_id_in in number,
   975:         type_in in char,
   976:         quantity_in in number,
   977:                 update_family_countsYN in number := 1
   978:     ) is
   979:         cursor usergroups is
   980:             select    user_id, user_group_id, ugt.label
   981:             from    rhnUserGroupType    ugt,
   982:                     rhnUserGroup        ug,
   983:                     rhnUserGroupMembers    ugm
   984:             where    1=1
   985:                 and ugm.user_group_id = group_id_in
   986:                 and ugm.user_id in (
   987:                     select    user_id
   988:                     from    (
   989:                         select    rownum row_number,
   990:                                 user_id,
   991:                                 time
   992:                         from    (
   993:                             select    user_id,
   994:                                     modified time
   995:                             from    rhnUserGroupMembers
   996:                             where    user_group_id = group_id_in
   997:                             order by time asc
   998:                         )
   999:                     )
  1000:                     where    row_number > quantity_in
  1001:                 )
  1002:                 and ugm.user_group_id = ug.id
  1003:                 and ug.group_type = ugt.id;
  1004:         cursor servergroups is
  1005:            select  server_id, server_group_id, sgt.id as group_type_id, sgt.label
  1006:             from    rhnServerGroupType              sgt,
  1007:                             rhnServerGroup                  sg,
  1008:                             rhnServerGroupMembers   sgm
  1009:             where   1=1
  1010:                     and sgm.server_group_id = group_id_in
  1011:                     and sgm.server_id in (
  1012:                             select  server_id
  1013:                             from    (
  1014:                                     select  rownum row_number,
  1015:                                                     server_id,
  1016:                                                     time
  1017:                                     from    (
  1018:                                             select  sep.server_id,
  1019:                                                     sep.modified time
  1020:                                             from
  1021:                                                 rhnServerEntitlementPhysical sep
  1022:                                             where

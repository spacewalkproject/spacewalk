 46 -def compile(file, cfile=None, dfile=None): 

 47      """Byte-compile one Python source file to Python bytecode. 
 48   
 49      Arguments: 
 50   
 51      file:  source filename 
 52      cfile: target filename; defaults to source with 'c' or 'o' appended 
 53             ('c' normally, 'o' in optimizing mode, giving .pyc or .pyo) 
 54      dfile: purported filename; defaults to source (this is the filename 
 55             that will show up in error messages) 
 56   
 57      Note that it isn't necessary to byte-compile Python modules for 
 58      execution efficiency -- Python itself byte-compiles a module when 
 59      it is loaded, and if it can, writes out the bytecode to the 
 60      corresponding .pyc (or .pyo) file. 
 61   
 62      However, if a Python installation is shared between users, it is a 
 63      good idea to byte-compile all modules upon installation, since 
 64      other users may not be able to write in the source directories, 
 65      and thus they won't be able to write the .pyc/.pyo file, and then 
 66      they would be byte-compiling every module each time it is loaded. 
 67      This can slow down program start-up considerably. 
 68   
 69      See compileall.py for a script/module that uses this module to 
 70      byte-compile all installed files (or all files in selected 
 71      directories). 
 72      """ 
 73   
 74      f = open(file) 
 75      try: 
 76          timestamp = long(os.fstat(f.fileno())[8]) 
 77      except AttributeError: 
 78          timestamp = long(os.stat(file)[8]) 
 79      codestring = f.read() 
 80      f.close() 
 81      if codestring and codestring[-1] != '\n': 
 82          codestring = codestring + '\n' 
 83      try: 
 84          codeobject = __builtin__.compile(codestring, dfile or file, 'exec') 
 85      except SyntaxError, detail: 
 86          lines = traceback.format_exception_only(SyntaxError, detail) 
 87          sys.stderr.write("%s: Error compiling\n" % file) 
 88          for line in lines:

 91          return -1

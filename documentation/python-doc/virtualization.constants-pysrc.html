51  ## 
52  # This structure maps the libvirt state enumeration to labels that RHN  
53  # understands. 
54  # Reasons we don't care about differences between NOSTATE, RUNNING and BLOCKED: 
55  # 1. technically, the domain is still "running" 
56  # 2. RHN / RHN Satellite / Spacewalk are not able to display 'blocked' & 'nostate' 
57  #    as valid states 
58  # 3. to avoid 'Abuse of Service' messages: bugs #230106 and #546676 
59   
60  VIRT_STATE_NAME_MAP = ( StateType.RUNNING,  # VIR_DOMAIN_NOSTATE 
61                          StateType.RUNNING,  # VIR_DOMAIN_RUNNING 
62                          StateType.RUNNING,  # VIR_DOMAIN_BLOCKED 
63                          StateType.PAUSED,   # VIR_DOMAIN_PAUSED 
64                          StateType.SHUTDOWN, # VIR_DOMAIN_SHUTDOWN 
65                          StateType.SHUTOFF,  # VIR_DOMAIN_SHUTOFF 
66                          StateType.CRASHED)  # VIR_DOMAIN_CRASHED

593              try: 
594                  activateSatellite_remote(options) 
595              except RHNCertRemoteActivationException, e: 
596                  writeError(e) 
597                  return 20 
598              except RHNCertRemoteNoManagementSlotsException, e: 
599                  writeError(e) 
600                  return 80 
601              except RHNCertRemoteSatelliteAlreadyActivatedException, e: 
602                  # note, this is normally a 1021 fault, but it's what we want 
603                  # so let's return 0 
604                  return 0 
605              except RHNCertRemoteNoAccessToSatChannelException, e: 
606                  writeError(e) 
607                  return 82 
608              except RHNCertRemoteInsufficientChannelEntitlementsException, e: 
609                  writeError(e) 
610                  return 83 
611              except RHNCertRemoteInvalidSatCertificateException, e: 
612                  writeError(e) 
613                  return 84 
614              except RHNCertRemoteSatelliteNotActivatedException, e: 
615                  writeError(e) 
616                  return 85 
617              except RHNCertRemoteSatelliteNoBaseChannelException, e: 
618                  writeError(e) 
619                  return 86 
620              except RHNCertNoSatChanForVersion, e: 
621                  writeError(e) 
622                  return 87 
623              except TimeoutException, e: 
624                  writeError(e) 
625                  return 89 
626   
627          # channel family stuff 
628          if not options.disconnected: 
629              try: 
630                  populateChannelFamilies(options) 
631              except PopulateChannelFamiliesException, e: 
632                  writeError(e) 
633                  return 40 
634          else: 
635              # We're disconnected so we'll only update the channels we 
636              # already know about. 
637              localUpdateChannels()
